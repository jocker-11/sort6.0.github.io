#include ＜stdio.h＞
int main(void){
int box[ 4 ], i, n, y ;
printf ("請輸入任意且互不相等的四個數字： \n") ;
for ( i= 0 ; i＜ 4 ; i ++ ) {
scanf ( "%d", &box[ i ] ) ;
}
for ( i= 0 ; i＜ 3 ; i ++ ) {
for ( n =0 ; n＜ 3 ; n ++ ) {
if ( box[ n ]＞ box[ n +1 ] ) {
y = box [ n +1 ] ;
box [ n +1 ] = box [ n ] ;
box [ n ] = y ;
}
}
}
for ( i= 0 ; i＜ 4 ; i ++ ) {
printf ( "%d", &box [ i ] ) ;
}

}
